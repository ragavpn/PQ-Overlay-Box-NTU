// The following commands are used to build the docker images and run the containers
// Quantum_container 1 and 2 are the substitue for the raspberry pi's being used in our network


# build docker images

sudo docker build -t quantum_container1 quantum_container1/
sudo docker build -t quantum_container2 quantum_container2/
sudo docker build -t machine_a machine_a/
sudo docker build -t machine_b machine_b/

# check output

sudo docker logs quantum_container1
sudo docker logs quantum_container2
sudo docker logs machine_a
sudo docker logs machine_b

# create network

sudo docker network create quantum_network

# run containers

sudo docker run -d --name quantum_container1 --network quantum_network quantum_container1
sudo docker run -d --name quantum_container2 --network quantum_network quantum_container2
sudo docker run -d --name machine_a --network quantum_network machine_a
sudo docker run -d --name machine_b --network quantum_network machine_b

# delete containers

sudo docker rm quantum_container1
sudo docker rm quantum_container2
sudo docker rm machine_a
sudo docker rm machine_b

# check ip address and stop containers

sudo docker stop machine_b
sudo docker inspect -f '{{range .NetworkSettings.Networks}}{{.IPAddress}}{{end}}' machine_b
